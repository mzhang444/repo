/**
* JavaScript commonly uses closures to define CONSTANTS. We use this module to define a unique implementation of such constructs. 
*
* Example of creating a constant
* 
* require(['constants'],function(CONSTANT){
* var q = CONSTANT(12);
* console.log(q);
* });
*
* @name CONSTANT
* @class Constant
* @param value for the new constant
* @returns returns a closure, that will return the initial constant value passe in at creation.
*/

define(function(){

	'use strict';

	return 	function(x){
		function returnConstant(){
			return x;
		}
		return returnConstant();
	};
});
