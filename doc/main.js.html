/**
 * 
 * The main entrypoint to the applicaiton. Here gets everything assembled
 * @name Main
 * @author Peter
 * @constructor
 * @param  Not important just a trick
 */
(function (root) {
	"use strict";
	require(["config"], function (config){

		require.config(config);

		require(["js/templates/precompiled/templatesModule.js"]);

		require(["App", "ember"], function (App, Ember) {
			
			var AppName = config.app_name || "App";
			root[AppName] = App = Ember.Application.create(App);
		});
	});
})(this);
